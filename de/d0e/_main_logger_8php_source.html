138  protected function send($message, $level = -1){
139  $now = time();
140  $message = TextFormat::toANSI(TextFormat::AQUA . date("H:i:s", $now) . TextFormat::RESET . " " . $message . TextFormat::RESET . PHP_EOL);
141  $cleanMessage = TextFormat::clean(preg_replace('/\x1b\[[0-9;]*m/', "", $message));
142 
143  if(!$this->hasANSI){
144  echo $cleanMessage;
145  }else{
146  echo $message;
147  }
148 
149  if($this->attachment instanceof \ThreadedLoggerAttachment){
150  $this->attachment->call($level, $message);
151  }
152 
153  $this->logStream .= date("Y-m-d", $now) . " " . $cleanMessage;
154  }

156  public function run(){
157  $this->shutdown = false;
158  $this->logResource = fopen($this->logFile, "a+b");
159  if(!is_resource($this->logResource)){
160  throw new \RuntimeException("Couldn't open log file");
161  }
162 
163  while($this->shutdown === false){
164  if(strlen($this->logStream) >= 4096){
165  $this->lock();
166  $chunks = strlen($this->logStream) >> 12;
167  $chunk = substr($this->logStream, 0, $chunks << 12);
168  $this->logStream = substr($this->logStream, $chunks << 12);
169  $this->unlock();
170  fwrite($this->logResource, $chunk);
171  }else{
172  usleep(250000); //sleep for 0.25 seconds
173  }
174  }
175  if(strlen($this->logStream) > 0){
176  fwrite($this->logResource, $this->logStream);
177  }
178 
179  fclose($this->logResource);
180  }
