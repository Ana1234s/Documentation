113  public function handlePackets(){
114  if(($len = @socket_read($this->socket, 4)) !== false and $len !== ""){
115  if(strlen($len) < 4){
116  $len .= $this->socketRead(4 - strlen($len));
117  }
118 
119  $packet = $this->socketRead(Binary::readInt($len));
120  $pid = ord($packet{0});
121  $offset = 1;
122 
123  if($pid === GenerationManager::PACKET_REQUEST_CHUNK){
124  $levelID = Binary::readInt(substr($packet, $offset, 4));
125  $offset += 4;
126  $chunkX = Binary::readInt(substr($packet, $offset, 4));
127  $offset += 4;
128  $chunkZ = Binary::readInt(substr($packet, $offset, 4));
129  $this->handleRequest($levelID, $chunkX, $chunkZ);
130  }elseif($pid === GenerationManager::PACKET_SEND_CHUNK){
131  $levelID = Binary::readInt(substr($packet, $offset, 4));
132  $offset += 4;
133  $len = ord($packet{$offset++});
135  $class = substr($packet, $offset, $len);
136  $offset += $len;
137  $level = $this->server->getLevel($levelID);
138  if($level instanceof Level){
139  $chunk = $class::fromBinary(substr($packet, $offset), $level->getProvider());
140  $this->receiveChunk($levelID, $chunk);
141  }
142  }
143  }
