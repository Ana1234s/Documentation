60  public static function getUniqueID($raw = false, $extra = ""){
61  $machine = php_uname("a");

63  $machine .= sys_get_temp_dir();
64  $machine .= $extra;
65  if(Utils::getOS() == "win"){
66  exec("ipconfig /ALL", $mac);
67  $mac = implode("\n", $mac);
68  if(preg_match_all("#Physical Address[. ]{1,}: ([0-9A-F\-]{17})#", $mac, $matches)){
69  foreach($matches[1] as $i => $v){

71  unset($matches[1][$i]);

73  }
74  $machine .= implode(" ", $matches[1]); //Mac Addresses
75  }
76  }
77  $data = $machine . PHP_MAXPATHLEN;
78  $data .= PHP_INT_MAX;
79  $data .= PHP_INT_SIZE;
80  $data .= get_current_user();
81  foreach(get_loaded_extensions() as $ext){
82  $data .= $ext . ":" . phpversion($ext);
83  }

211  public static function getRandomBytes($length = 16, $secure = true, $raw = true, $startEntropy = "", &$rounds = 0, &$drop = 0){
212  static $lastRandom = "";
213  $output = "";
214  $length = abs((int) $length);
215  $secureValue = "";
216  $rounds = 0;
217  $drop = 0;
218  while(!isset($output{$length - 1})){
219  //some entropy, but works ^^
220  $weakEntropy = array(
221  is_array($startEntropy) ? implode($startEntropy) : $startEntropy,
222  serialize(@stat(__FILE__)),
223  __DIR__,
224  PHP_OS,
225  microtime(),
226  (string) lcg_value(),
227  (string) PHP_MAXPATHLEN,
228  PHP_SAPI,
229  (string) PHP_INT_MAX . "." . PHP_INT_SIZE,
230  serialize($_SERVER),
231  serialize(get_defined_constants()),
232  get_current_user(),
233  serialize(ini_get_all()),
234  (string) memory_get_usage() . "." . memory_get_peak_usage(),
235  php_uname(),
236  phpversion(),
237  extension_loaded("gmp") ? gmp_strval(gmp_random(4)) : microtime(),
238  zend_version(),
239  (string) getmypid(),
240  (string) getmyuid(),
241  (string) mt_rand(),
242  (string) getmyinode(),
243  (string) getmygid(),
244  (string) rand(),
245  function_exists("zend_thread_id") ? ((string) zend_thread_id()) : microtime(),
246  function_exists("getrusage") ? @implode(getrusage()) : microtime(),
247  function_exists("sys_getloadavg") ? @implode(sys_getloadavg()) : microtime(),
248  serialize(get_loaded_extensions()),
249  sys_get_temp_dir(),
250  (string) disk_free_space("."),
251  (string) disk_total_space("."),
252  uniqid(microtime(), true),
253  file_exists("/proc/cpuinfo") ? file_get_contents("/proc/cpuinfo") : microtime(),
254  );
255 
256  shuffle($weakEntropy);
257  $value = hash("sha512", implode($weakEntropy), true);
258  $lastRandom .= $value;
259  foreach($weakEntropy as $k => $c){ //mixing entropy values with XOR and hash randomness extractor
260  $value ^= hash("sha256", $c . microtime() . $k, true) . hash("sha256", mt_rand() . microtime() . $k . $c, true);
261  $value ^= hash("sha512", ((string) lcg_value()) . $c . microtime() . $k, true);
262  }
263  unset($weakEntropy);
264 
265  if($secure === true){
266  $strongEntropyValues = array(
267  is_array($startEntropy) ? hash("sha512", $startEntropy[($rounds + $drop) % count($startEntropy)], true) : hash("sha512", $startEntropy, true), //Get a random index of the startEntropy, or just read it
268  file_exists("/dev/urandom") ? fread(fopen("/dev/urandom", "rb"), 64) : str_repeat("\x00", 64),
269  (function_exists("openssl_random_pseudo_bytes") and version_compare(PHP_VERSION, "5.3.4", ">=")) ? openssl_random_pseudo_bytes(64) : str_repeat("\x00", 64),
270  function_exists("mcrypt_create_iv") ? mcrypt_create_iv(64, MCRYPT_DEV_URANDOM) : str_repeat("\x00", 64),
271  $value,
272  );
273  $strongEntropy = array_pop($strongEntropyValues);
274  foreach($strongEntropyValues as $value){
275  $strongEntropy = $strongEntropy ^ $value;
276  }
277  $value = "";
278  //Von Neumann randomness extractor, increases entropy
279  $bitcnt = 0;
280  for($j = 0; $j < 64; ++$j){
281  $a = ord($strongEntropy{$j});
282  for($i = 0; $i < 8; $i += 2){
283  $b = ($a & (1 << $i)) > 0 ? 1 : 0;
284  if($b != (($a & (1 << ($i + 1))) > 0 ? 1 : 0)){
285  $secureValue |= $b << $bitcnt;
286  if($bitcnt == 7){
287  $value .= chr($secureValue);
288  $secureValue = 0;
289  $bitcnt = 0;
290  }else{
291  ++$bitcnt;
292  }
293  ++$drop;
294  }else{
295  $drop += 2;
296  }
297  }
298  }
299  }
300  $output .= substr($value, 0, min($length - strlen($output), $length));
301  unset($value);
302  ++$rounds;
303  }
304  $lastRandom = hash("sha512", $lastRandom, true);
305 
306  return $raw === false ? bin2hex($output) : $output;
307  }
