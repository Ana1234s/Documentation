158  if(($len = $this->socket->readPacket($buffer, $source, $port)) > 0){
159  $this->receiveBytes += $len;

219  public function receiveStream(){
220  if(($len = @socket_read($this->internalSocket, 4)) !== false){
221  if(strlen($len) < 4){
222  $len .= $this->socketRead(4 - strlen($len));
223  }
224  $packet = $this->socketRead(Binary::readInt($len));
225  $id = ord($packet{0});
226  $offset = 1;
227  if($id === RakLib::PACKET_ENCAPSULATED){
228  $len = ord($packet{$offset++});
229  $identifier = substr($packet, $offset, $len);
230  $offset += $len;
231  if(isset($this->sessions[$identifier])){
232  $flags = ord($packet{$offset++});
233  $buffer = substr($packet, $offset);
234  $this->sessions[$identifier]->addEncapsulatedToQueue(EncapsulatedPacket::fromBinary($buffer, true), $flags);
235  }else{
236  $this->streamInvalid($identifier);
237  }
238  }elseif($id === RakLib::PACKET_CLOSE_SESSION){
239  $len = ord($packet{$offset++});
240  $identifier = substr($packet, $offset, $len);
241  if(isset($this->sessions[$identifier])){
242  $this->removeSession($this->sessions[$identifier]);
243  }else{
244  $this->streamInvalid($identifier);
245  }
246  }elseif($id === RakLib::PACKET_INVALID_SESSION){
247  $len = ord($packet{$offset++});
248  $identifier = substr($packet, $offset, $len);
249  if(isset($this->sessions[$identifier])){
250  $this->removeSession($this->sessions[$identifier]);
251  }
252  }elseif($id === RakLib::PACKET_SET_OPTION){
253  $len = ord($packet{$offset++});
254  $name = substr($packet, $offset, $len);
255  $offset += $len;
256  $value = substr($packet, $offset);
257  switch($name){
258  case "name":
259  $this->name = $value;
260  break;
261  }
262  }elseif($id === RakLib::PACKET_SHUTDOWN){
263  foreach($this->sessions as $session){
264  $this->removeSession($session);
265  }
266  @socket_close($this->internalSocket);
267  $this->socket->close();
268  $this->shutdown = true;
269  }elseif($id === RakLib::PACKET_EMERGENCY_SHUTDOWN){
270  $this->shutdown = true;
271  }
272 
273  return true;
274  }
275 
276  return false;
277  }
