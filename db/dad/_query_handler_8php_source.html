59  public function regenerateInfo(){
60  $str = "";
61  $plist = "PocketMine-MP " . $this->server->getPocketMineVersion();
62  $pl = $this->server->getPluginManager()->getPlugins();
63  if(count($pl) > 0 and $this->server->getProperty("settings.query-plugins", true) === true){
64  $plist .= ":";
65  foreach($pl as $p){
66  $d = $p->getDescription();
67  $plist .= " " . str_replace(array(";", ":", " "), array("", "", "_"), $d->getName()) . " " . str_replace(array(";", ":", " "), array("", "", "_"), $d->getVersion()) . ";";
68  }
69  $plist = substr($plist, 0, -1);
70  }
71  $KVdata = array(
72  "splitnum" => chr(128),
73  "hostname" => $this->server->getServerName(),
74  "gametype" => ($this->server->getGamemode() & 0x01) === 0 ? "SMP" : "CMP",
75  "game_id" => "MINECRAFTPE",
76  "version" => $this->server->getVersion(),
77  "server_engine" => $this->server->getName() . " " . $this->server->getPocketMineVersion(),
78  "plugins" => $plist,
79  "map" => $this->server->getDefaultLevel() === null ? "unknown" : $this->server->getDefaultLevel()->getName(),
80  "numplayers" => count($this->server->getOnlinePlayers()),
81  "maxplayers" => $this->server->getMaxPlayers(),
82  "whitelist" => $this->server->hasWhitelist() === true ? "on" : "off",
83  "hostport" => $this->server->getPort()
84  );
85  foreach($KVdata as $key => $value){
86  $str .= $key . "\x00" . $value . "\x00";
87  }
88  $str .= "\x00\x01player_\x00\x00";
89  foreach($this->server->getOnlinePlayers() as $player){
90  if($player->getName() != ""){
91  $str .= $player->getName() . "\x00";
92  }
93  }
94  $str .= "\x00";
95  $this->longData = $str;
96  $this->timeout = microtime(true) + 5;
97  }

108  public function handle($address, $port, $packet){
109  $offset = 2;
110  $packetType = ord($packet{$offset++});
111  $sessionID = Binary::readInt(substr($packet, $offset, 4));
112  $offset += 4;
113  $payload = substr($packet, $offset);
114 
115  switch($packetType){
116  case self::HANDSHAKE: //Handshake
117  $reply = chr(self::HANDSHAKE);
118  $reply .= Binary::writeInt($sessionID);
119  $reply .= self::getTokenString($this->token, $address) . "\x00";
120 
121  $this->server->sendPacket($address, $port, $reply);
122  break;
123  case self::STATISTICS: //Stat
124  $token = Binary::readInt(substr($payload, 0, 4));
125  if($token !== self::getTokenString($this->token, $address) and $token !== self::getTokenString($this->lastToken, $address)){
126  break;
127  }
128  $reply = chr(self::STATISTICS);
129  $reply .= Binary::writeInt($sessionID);
130  if(strlen($payload) === 8){
131  if($this->timeout < microtime(true)){
132  $this->regenerateInfo();
133  }
134  $reply .= $this->longData;
135  }else{
136  $reply .= $this->server->getServerName() . "\x00" . (($this->server->getGamemode() & 0x01) === 0 ? "SMP" : "CMP") . "\x00" . ($this->server->getDefaultLevel() === null ? "unknown" : $this->server->getDefaultLevel()->getName()) . "\x00" . count($this->server->getOnlinePlayers()) . "\x00" . $this->server->getMaxPlayers() . "\x00" . Binary::writeLShort($this->server->getPort()) . $this->server->getIp() . "\x00";
137  }
138  $this->server->sendPacket($address, $port, $reply);
139  break;
140  }
141  }
