34  public function __construct($version = \pocketmine\VERSION){
35  if(is_int($version)){
36  $this->minor = $version & 0x1F;
37  $this->major = ($version >> 5) & 0x0F;
38  $this->generation = ($version >> 9) & 0x0F;
39  }else{
40  $version = preg_split("/([A-Za-z]*)[ _\\-]?([0-9]*)\\.([0-9]*)\\.{0,1}([0-9]*)(dev|)(-[\\0-9]{1,}|)/", $version, -1, PREG_SPLIT_DELIM_CAPTURE);
41  $this->generation = isset($version[2]) ? (int) $version[2] : 0; //0-15
42  $this->major = isset($version[3]) ? (int) $version[3] : 0; //0-15
43  $this->minor = isset($version[4]) ? (int) $version[4] : 0; //0-31
44  $this->development = $version[5] === "dev" ? true : false;
45  if($version[6] !== ""){
46  $this->build = intval(substr($version[6], 1));
47  }else{
48  $this->build = 0;
49  }
50  }
