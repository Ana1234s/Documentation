129  $i = ($x << 11) + ($z << 7) + $y;
130 
131  $changed = false;
132 
133  if($blockId !== null){
134  $blockId = chr($blockId);

136  $this->blocks{$i} = $blockId;

142  $i >>= 1;

144  if(($y & 1) === 0){
145  $this->data{$i} = chr(($old_m & 0xf0) | ($meta & 0x0f));
146  if(($old_m & 0x0f) !== $meta){
147  $changed = true;
148  }
149  }else{

151  if((($old_m & 0xf0) >> 4) !== $meta){
152  $changed = true;
153  }
