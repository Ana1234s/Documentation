80  public function handlePacket(){
81  if(($len = @socket_read($this->socket, 4)) !== false){
82  if(strlen($len) < 4){
83  $len .= $this->socketRead(4 - strlen($len));
84  }
85  $packet = $this->socketRead(Binary::readInt($len));
86  $id = ord($packet{0});
87  $offset = 1;
88  if($id === RakLib::PACKET_ENCAPSULATED){
89  $len = ord($packet{$offset++});
90  $identifier = substr($packet, $offset, $len);
91  $offset += $len;
92  $flags = ord($packet{$offset++});
93  $buffer = substr($packet, $offset);
94  $this->instance->handleEncapsulated($identifier, EncapsulatedPacket::fromBinary($buffer, true), $flags);
95  }elseif($id === RakLib::PACKET_OPEN_SESSION){
96  $len = ord($packet{$offset++});
97  $identifier = substr($packet, $offset, $len);
98  $offset += $len;
99  $len = ord($packet{$offset++});
100  $address = substr($packet, $offset, $len);
101  $offset += $len;
102  $port = Binary::readShort(substr($packet, $offset, 2), false);
103  $offset += 2;
104  $clientID = Binary::readLong(substr($packet, $offset, 8));
105  $this->instance->openSession($identifier, $address, $port, $clientID);
106  }elseif($id === RakLib::PACKET_CLOSE_SESSION){
107  $len = ord($packet{$offset++});
108  $identifier = substr($packet, $offset, $len);
109  $offset += $len;
110  $len = ord($packet{$offset++});
111  $reason = substr($packet, $offset, $len);
112  $this->instance->closeSession($identifier, $reason);
113  }elseif($id === RakLib::PACKET_INVALID_SESSION){
114  $len = ord($packet{$offset++});
115  $identifier = substr($packet, $offset, $len);
116  $this->instance->closeSession($identifier, "Invalid session");
117  }elseif($id === RakLib::PACKET_ACK_NOTIFICATION){
118  $len = ord($packet{$offset++});
119  $identifier = substr($packet, $offset, $len);
120  $offset += $len;
121  $identifierACK = Binary::readInt(substr($packet, $offset, 4));
122  $this->instance->notifyACK($identifier, $identifierACK);
123  }
