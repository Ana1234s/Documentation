114  public static function readMetadata($value, $types = false){
115  $offset = 0;
116  $m = [];
117  $b = ord($value{$offset});
118  ++$offset;
119  while($b !== 127 and isset($value{$offset})){
120  $bottom = $b & 0x1F;
121  $type = $b >> 5;
122  switch($type){
123  case 0:
124  $r = self::readByte($value{$offset});
125  ++$offset;
126  break;
127  case 1:
128  $r = self::readLShort(substr($value, $offset, 2));
129  $offset += 2;

133  $offset += 4;
134  break;
135  case 3:

139  case 4:
