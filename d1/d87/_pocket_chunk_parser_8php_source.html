110  public function parseChunk($X, $Z){

112  $Z = (int) $Z;
113  $offset = $this->getOffset($X, $Z);
114  $len = Binary::readLInt(substr($this->raw, $offset, 4));
115  $offset += 4;
116  $chunk = [
117  0 => [], //Block
118  1 => [], //Data
119  2 => [], //SkyLight
