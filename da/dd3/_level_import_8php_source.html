37  public function import(){
38  if(file_exists($this->path . "tileEntities.dat")){ //OldPM
39  $level = unserialize(file_get_contents($this->path . "level.dat"));
40  MainLogger::getLogger()->info("Importing OldPM level \"" . $level["LevelName"] . "\" to PMF format");
41  $entities = new Config($this->path . "entities.yml", Config::YAML, unserialize(file_get_contents($this->path . "entities.dat")));
42  $entities->save();
43  $tiles = new Config($this->path . "tiles.yml", Config::YAML, unserialize(file_get_contents($this->path . "tileEntities.dat")));
44  $tiles->save();
45  }elseif(file_exists($this->path . "chunks.dat") and file_exists($this->path . "level.dat")){ //Pocket
46  $nbt = new NBT(NBT::LITTLE_ENDIAN);
47  $nbt->read(substr(file_get_contents($this->path . "level.dat"), 8));
48  $level = $nbt->getData();
49  if($level["LevelName"] == ""){
50  $level["LevelName"] = "world" . time();
51  }
52  MainLogger::getLogger()->info("Importing Pocket level \"" . $level->LevelName . "\" to PMF format");
53  unset($level->Player);
54  $nbt->read(substr(file_get_contents($this->path . "entities.dat"), 12));
55  $entities = $nbt->getData();
56  if(!isset($entities->TileEntities)){
57  $entities->TileEntities = [];
58  }
59  $tiles = $entities->TileEntities;
60  $entities = $entities->Entities;
61  $entities = new Config($this->path . "entities.yml", Config::YAML, $entities);
62  $entities->save();
63  $tiles = new Config($this->path . "tiles.yml", Config::YAML, $tiles);
64  $tiles->save();
65  }else{
66  return false;
67  }
68 
69  $pmf = new LevelFormat($this->path . "level.pmf", array(
70  "name" => $level->LevelName,
71  "seed" => $level->RandomSeed,
72  "time" => $level->Time,
73  "spawnX" => $level->SpawnX,
74  "spawnY" => $level->SpawnY,
75  "spawnZ" => $level->SpawnZ,
76  "height" => 8,
77  "generator" => "default",
78  "generatorSettings" => "",
79  "extra" => ""
80  ));
81  $chunks = new PocketChunkParser();
82  $chunks->loadFile($this->path . "chunks.dat");
83  $chunks->loadMap();
84  for($Z = 0; $Z < 16; ++$Z){
85  for($X = 0; $X < 16; ++$X){
86  $chunk = array(
87  0 => "",
88  1 => "",
89  2 => "",
90  3 => "",
91  4 => "",
92  5 => "",
93  6 => "",
94  7 => ""
95  );
96 
97  $pmf->initCleanChunk($X, $Z);
98  for($z = 0; $z < 16; ++$z){
99  for($x = 0; $x < 16; ++$x){
100  $block = $chunks->getChunkColumn($X, $Z, $x, $z, 0);
101  $meta = $chunks->getChunkColumn($X, $Z, $x, $z, 1);
102  for($Y = 0; $Y < 8; ++$Y){
103  $chunk[$Y] .= substr($block, $Y << 4, 16);
104  $chunk[$Y] .= substr($meta, $Y << 3, 8);
105  $chunk[$Y] .= "\x00\x00\x00\x00\x00\x00\x00\x00";
106  }
107  }
108  }
109  foreach($chunk as $Y => $data){
110  $pmf->setMiniChunk($X, $Z, $Y, $data);
111  }
112  $pmf->setPopulated($X, $Z);
113  $pmf->saveChunk($X, $Z);
114  }
115  MainLogger::getLogger()->notice("Importing level " . ceil(($Z + 1) / 0.16) . "%");
116  }
117  $chunks->map = null;
118  $chunks = null;
119  @unlink($this->path . "level.dat");
120  @unlink($this->path . "level.dat_old");
121  @unlink($this->path . "player.dat");
122  @unlink($this->path . "entities.dat");
123  @unlink($this->path . "chunks.dat");
124  @unlink($this->path . "chunks.dat.gz");
125  @unlink($this->path . "tiles.dat");
126  unset($chunks, $level, $entities, $tiles, $nbt);
127 
128  return true;
129  }
