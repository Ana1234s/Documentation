47  public static function fromBinary($binary, $internal = false, &$offset = null){
48  $packet = new EncapsulatedPacket;
49  $flags = ord($binary{0});
50  $packet->reliability = $reliability = ($flags & 0b11100000) >> 5;
51  $packet->hasSplit = $hasSplit = ($flags & 0b0010000) > 0;
52  if($internal){
53  $length = Binary::readInt(substr($binary, 1, 4));
54  $packet->identifierACK = Binary::readInt(substr($binary, 5, 4));
55  $offset = 9;
56  }else{
57  $length = (int) ceil(Binary::readShort(substr($binary, 1, 2), false) / 8);
58  $offset = 3;
59  }
60 
61 
62  /*
63  * From http://www.jenkinssoftware.com/raknet/manual/reliabilitytypes.html
64  *
65  * Default: 0b010 (2) or 0b011 (3)
66  *
67  * 0: UNRELIABLE
68  * 1: UNRELIABLE_SEQUENCED
69  * 2: RELIABLE
70  * 3: RELIABLE_ORDERED
71  * 4: RELIABLE_SEQUENCED
72  * 5: UNRELIABLE_WITH_ACK_RECEIPT
73  * 6: RELIABLE_WITH_ACK_RECEIPT
74  * 7: RELIABLE_ORDERED_WITH_ACK_RECEIPT
75  */
76 
77  if(
78  $reliability === 2 or
79  $reliability === 3 or
80  $reliability === 4 or
81  $reliability === 6 or
82  $reliability === 7
83  ){
84  $packet->messageIndex = Binary::readTriad(strrev(substr($binary, $offset, 3)));
85  $offset += 3;
86  }
87 
88  if(
89  $reliability === 1 or
90  $reliability === 3 or
91  $reliability === 4 or
92  $reliability === 7
93  ){
94  $packet->orderIndex = Binary::readTriad(strrev(substr($binary, $offset, 3)));
95  $offset += 3;
96  $packet->orderChannel = ord($binary{$offset++});
97  }
98 
99  if($hasSplit){
100  $packet->splitCount = Binary::readInt(substr($binary, $offset, 4));
101  $offset += 4;
102  $packet->splitID = Binary::readShort(substr($binary, $offset, 2));
103  $offset += 2;
104  $packet->splitIndex = Binary::readInt(substr($binary, $offset, 4));
105  $offset += 4;
106  }
107 
108  $packet->buffer = substr($binary, $offset, $length);
109  $offset += $length;
110 
111  return $packet;
112  }
