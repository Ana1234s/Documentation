495  $randRange = $randRange > $this->chunkTickRadius ? $this->chunkTickRadius : $randRange;
496

498  $x = $player->x >> 4;
499  $z = $player->x >> 4;
500 
501  $index = Level::chunkHash($x, $z);
502  $existingPlayers = max(0, isset($this->chunkTickList[$index]) ? $this->chunkTickList[$index] : 0);
503  $this->chunkTickList[$index] = $existingPlayers + 1;
504  for($chunk = 0; $chunk < $chunksPerPlayer; ++$chunk){
505  $dx = mt_rand(-$randRange, $randRange);
506  $dz = mt_rand(-$randRange, $randRange);
507  $hash = Level::chunkHash($dx + $x, $dz + $z);

512  }

515

517  Level::getXZ($index, $chunkX, $chunkZ);
518

520  unset($this->chunkTickList[$index]);
521  continue;

524 
525

528  if(!($section instanceof EmptyChunkSection)){

530  $k = mt_rand(0, PHP_INT_MAX);
531  for($i = 0; $i < 3; ++$i){

533  $x = $j & 0x0f;
534  $y = ($j >> 8) & 0x0f;
535  $z = ($j >> 16) & 0x0f;
536  $k %= 1073741827;
