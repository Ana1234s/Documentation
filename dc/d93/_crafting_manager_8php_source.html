366  public function matchTransaction(CraftingTransactionGroup $ts){
367  $result = $ts->getResult();
368 
369  if(!($result instanceof Item)){
370  return false;
371  }
372  $k = $result->getID() . ":" . $result->getDamage();
373 
374  if(!isset($this->recipeLookup[$k])){
375  return false;
376  }
377  $hash = "";
378  $input = $ts->getRecipe();
379  usort($input, array($this, "sort"));
380  $inputCount = 0;
381  foreach($input as $item){
382  $inputCount += $item->getCount();
383  $hash .= $item->getID() . ":" . ($item->getDamage() === null ? "?" : $item->getDamage()) . "x" . $item->getCount() . ",";
384  }
385  if(!isset($this->recipeLookup[$k][$hash])){
386  $hasRecipe = null;
387  foreach($this->recipeLookup[$k] as $recipe){
388  if($recipe instanceof ShapelessRecipe){
389  if($recipe->getIngredientCount() !== $inputCount){
390  continue;
391  }
392  $checkInput = $recipe->getIngredientList();
393  foreach($input as $item){
394  $amount = $item->getCount();
395  foreach($checkInput as $k => $checkItem){
396  if($checkItem->equals($item, $checkItem->getDamage() === null ? false : true)){
397  $remove = min($checkItem->getCount(), $amount);
398  $checkItem->setCount($checkItem->getCount() - $remove);
399  if($checkItem->getCount() === 0){
400  unset($checkInput[$k]);
401  }
402  $amount -= $remove;
403  if($amount === 0){
404  break;
405  }
406  }
407  }
408  }
409 
410  if(count($checkInput) === 0){
411  $hasRecipe = $recipe;
412  break;
413  }
414  }
415  if($hasRecipe instanceof Recipe){
416  break;
417  }
418  }
419 
420  if($hasRecipe === null){
421  return false;
422  }
423 
424  $recipe = $hasRecipe;
425  }else{
426  $recipe = $this->recipeLookup[$k][$hash];
427  }
428 
429  $checkResult = $recipe->getResult();
430  if($checkResult->equals($result, true) and $checkResult->getCount() === $result->getCount()){
431  return $recipe;
432  }
